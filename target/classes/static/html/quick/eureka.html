Eureka快速建站指引

Step1：Maven依赖

<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-netflix-eureka-server</artifactId>
</dependency>

Step2：配置

服务端配置
以使用SpringBoot为基础，配置在application.yml为例

server:
  port: 8761
#Eureka的配置
eureka:
  instance:
    hostname: localhost #Eureka服务端的实例名字
  client:
    register-with-eureka: false #表示是否向eureka注册中心注册自己
    fetch-registry: false #表示是否从 Eureka-Server 中获取服务注册信息 如果为false那么表示自己为注册中心
    service-url: #Eureka的监控页面地址
    defaultZone: http://${eureka.instance.hostname}:${server.port}/eureka/ #满足格式https://主机名:端口名/eureka/ 的格式就行

客户端配置
以使用SpringBoot为基础，配置在application.yml为例

eureka:
  instance:
     appname: ljw测试服务器 #注册在Eureka中的名字

Step3：SpringBoot启动类

服务端配置
以使用SpringBoot为例

@SpringBootApplication
@EnableEurekaServer
public class EurekaMain {
    public static void main(String[] args) {
        SpringApplication.run(EurekaMain.class, args);
    }
}

客户端配置
以使用SpringBoot为例

@SpringBootApplication
@EnableDiscoveryClient
public class ApplicationMain {
    public static void main(String[] args) {
         SpringApplication.run(ApplicationMain.class, args);
    }
}
